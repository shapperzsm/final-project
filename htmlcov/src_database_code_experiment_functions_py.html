65def who_is_playing(num_of_opponents, long_run_strategies=False): 

66 """ 

67 A function to choose which strategies will be playing against the Defector, where: 

68 

69 'num_of_opponents' is a numeric variable which states how many players will  

70 be competing (EXCLUDING the Defector); and 

71 

72 'long_run_strategies' is a Boolean variable which states whether strategies  

73 which have a long running time should be included in the competitors or  

74 not. It has a default value of False. 

75 

76 A list containing the selected strategies is returned. 

77 """ 

78 

79 if long_run_strategies == True: 

80 filterstrategies = { 

81 "manipulates_state": False, 

82 "manipulates_source": False, 

83 "inspects_source": False, 

84 } 

85 else: 

86 filterstrategies = { 

87 "long_run_time": False, 

88 "manipulates_state": False, 

89 "manipulates_source": False, 

90 "inspects_source": False, 

91 } 

92 

93 filtered_strategies = axl.filtered_strategies(filterstrategies) 

94 filtered_strategies.remove(axl.Defector) 

95 

96 opponent_strategies = random.sample(filtered_strategies, num_of_opponents) 

97 list_of_players = [opponent() for opponent in opponent_strategies] 

98 list_of_players.append(axl.Defector()) 

99 

100 return list_of_players

104def get_game(tournament_rep, player_list, prob_of_game_ending, noise): 

105 

106 """ 

107 A function which runs a tournament of A Prisoner's Dilemma and returns a dictionary containing the following: 'payoff matrix obtained' (an output from the Axelrod tournament function), 'number of tournament repeats' (from 'tournament_rep'), 'probability of game ending' (from 'prob_of_game_ending) and 'noise' (from 'noise'), where: 

108 

109 'tournament_rep' is a numeric variable stating how many times the tournament should be executed; 

110 

111 'player_list' is a list containing instances of specific strategy classes obtained from the Axelrod library; 

112 

113 'prob_of_game_ending' is a numeric variable between 0 and 1 which states the probability of the tournament ending on any specific turn; and 

114 

115 'noise' is a numeric variable between 0 and 1 as described on https://axelrod.readthedocs.io/en/stable/tutorials/further_topics/noisy_tournaments.html and https://axelrod.readthedocs.io/en/stable/reference/glossary.html#noise. 

116 """ 

117 

118 tournament = axl.Tournament( 

119 player_list, 

120 prob_end=prob_of_game_ending, 

121 repetitions=tournament_rep, 

122 noise=noise, 

123 ) 

124 tournament_results = tournament.play(progress_bar=False) 

125 

126 mean_payoff_matrix = np.array(tournament_results.payoff_matrix) 

127 

128 get_game_output_dict = { 

129 "payoff matrix obtained": mean_payoff_matrix, 

130 "number of tournament repeats": tournament_rep, 

131 "probability of game ending": prob_of_game_ending, 

132 "noise": noise, 

133 } 

134 return get_game_output_dict

138def get_prob_of_defection(payoff_matrix, support_enumeration=True): 

139 

140 """ 

141 A function which computes the Nash Equilibria of the game using one of three algorithms and returns a dictionary of the Nash Equilibria, the maximum and minimum probabilities of defection within the equilibria and whether the game could be degenerate. The input variables are: 

142 

143 'payoff_matrix', a numpy array containing the payoffs obtained for each action of the game; and 

144 

145 'support_enumeration', a boolean variable stating whether the support enumeration algorithm (if evaluated true) or the vertex enumeration algorithm (if evaluated False) is used to calculate the Nash equilibria. 

146 """ 

147 

148 game = nash.Game(payoff_matrix, payoff_matrix.transpose()) 

149 

150 with warnings.catch_warnings(record=True) as w: 

151 warnings.simplefilter("always") 

152 

153 if support_enumeration == True: 

154 nash_equilibria = list(game.support_enumeration()) 

155 

156 else: 

157 highlight_numpy_warning = np.seterr(all="warn") 

158 nash_equilibria = list(game.vertex_enumeration()) 

159 

160 if len(w) == 0: 

161 warning_message = None 

162 

163 else: 

164 warning_message = str([w[i].message for i in range(len(w))]) 

165 

166 if (len(nash_equilibria) == 0) or ("-Inf" in nash_equilibria): 

167 nash_equilibria = None 

168 prob_of_defection_in_equilibria = None 

169 least_prob_of_defection_in_equilibria = None 

170 greatest_prob_of_defection_in_equilibria = None 

171 

172 else: 

173 prob_of_defection_in_equilibria = [ 

174 sigma_1[-1] for sigma_1, _ in nash_equilibria 

175 ] 

176 

177 least_prob_of_defection_in_equilibria = min(prob_of_defection_in_equilibria) 

178 greatest_prob_of_defection_in_equilibria = max(prob_of_defection_in_equilibria) 

179 

180 get_prob_of_defect_output_dict = { 

181 "nash equilibria": np.array(nash_equilibria), 

182 "least prob of defect": least_prob_of_defection_in_equilibria, 

183 "greatest prob of defect": greatest_prob_of_defection_in_equilibria, 

184 "warning message": str(warning_message), 

185 } 

186 

187 return get_prob_of_defect_output_dict

304 max_num_of_opponents, 

305 number_of_player_samples, 

306 noise_probs,

309 database_filepath,

313): 

314 

315 """ 

316 A function which runs the experiment and writes the results to a database, where: 

317 

318 'max_num_of_opponents' is a numeric variable stating the largest number of players who will be competing against the Defector; 

319 

320 'number_of_player_samples' is a numeric variable indicating how many tournaments with the same number of players should be executed (for each value of noise and each probability of defection); 

321 

322 'noise_probs' is a list containing numeric variables between 0 and 1 indicating the amount of noise to be included in a tournament; 

323 

324 'game_ending_probs' is a list containing numeric variables between 0 and 1 indicating the probability of a game ending on any particular turn; 

325 

326 'tournament_rep' is a numeric variable stating the number of times each distinct tournament should be repeated (allows for the 'smoothing' of the mean payoffs obtained); 

327 

328 'database_filepath' is a string containing the relative path where the database (main.db) file is based; 

329 

330 'support_enumeration' is a boolean variable stating whether the support enumeration algorithm (if evaluated true) or the vertex enumeration algorithm (if evaluated False) is used to calculate the Nash equilibria; 

331  

332 'unique_tournament_identifier' is a numeric variable (default zero) which is the unique seed for a particular player set, at a particular noise level for a particular probability of the game ending; and 

333  

334 'tournament_player_set' is a numeric variable identifying a particular set of players. 

335 """ 

336 axl.seed(unique_tournament_identifier) 

337 while True: 

338 for num_of_opponents in range(1, max_num_of_opponents + 1): 

339 

340 for player_sample_repetition in range(1, number_of_player_samples + 1): 

341 players = who_is_playing( 

342 num_of_opponents=num_of_opponents, long_run_strategies=False 

343 ) 

344 

345 for noise in noise_probs: 

346 

347 for probability in game_ending_probs: 

348 tournament_run = get_game( 

349 tournament_rep=tournament_rep, 

350 player_list=players, 

351 prob_of_game_ending=probability, 

352 noise=noise, 

353 ) 

354 defection_probs = get_prob_of_defection( 

355 payoff_matrix=tournament_run["payoff matrix obtained"], 

356 support_enumeration=support_enumeration, 

357 ) 

358 

359 for player in players: 

360 write_record( 

361 experiment_number=unique_tournament_identifier, 

362 number_of_players=len(players), 

363 tournament_player_set=tournament_player_set, 

364 player_strategy_name=player, 

365 is_long_run_time=player.classifier["long_run_time"], 

366 is_stochastic=player.classifier["stochastic"], 

367 memory_depth_of_strategy=player.classifier[ 

368 "memory_depth" 

369 ], 

370 prob_of_game_ending=tournament_run[ 

371 "probability of game ending"

373 payoff_matrix=tournament_run["payoff matrix obtained"], 

374 num_of_repetitions=tournament_run[ 

375 "number of tournament repeats" 

376 ], 

377 nash_equilibria=defection_probs["nash equilibria"], 

378 least_prob_of_defection=defection_probs[ 

379 "least prob of defect" 

380 ],

382 "greatest prob of defect"

384 noise=tournament_run["noise"],
