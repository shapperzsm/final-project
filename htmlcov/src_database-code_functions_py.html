21def who_is_playing(num_of_opponents, seed, long_run_strategies=False): 

22 """ 

23 A function to choose which strategies will be playing against the Defector, where: 

24 

25 'num_of_opponents' is a numeric variable which states how many players will  

26 be competing (EXCLUDING the Defector); 

27 

28 'seed' is a numeric variable which ensures reproducibility if the same value is used; and 

29 

30 'long_run_strategies' is a Boolean variable which states whether strategies  

31 which have a long running time should be included in the competitors or  

32 not. It has a default value of False. 

33 

34 A list containing the selected strategies is returned. 

35 """ 

36 

37 if long_run_strategies == True: 

38 filterstrategies = { 

39 "manipulates_state": False, 

40 "manipulates_source": False, 

41 "inspects_source": False, 

42 } 

43 else: 

44 filterstrategies = { 

45 "long_run_time": False, 

46 "manipulates_state": False, 

47 "manipulates_source": False, 

48 "inspects_source": False, 

49 } 

50 

51 filtered_strategies = axl.filtered_strategies(filterstrategies) 

52 filtered_strategies.remove(axl.Defector) 

53 

54 axl.seed(seed) 

55 opponent_strategies = random.sample(filtered_strategies, num_of_opponents) 

56 list_of_players = [opponent() for opponent in opponent_strategies] 

57 list_of_players.append(axl.Defector()) 

58 

59 return list_of_players

99def get_prob_of_defection(payoff_matrix, nash_equilibrium_algorithm): 

100 

101 """ 

102 A function which computes the Nash Equilibria of the game using one of three algorithms and returns a dictionary of the Nash Equilibria, the maximum and minimum probabilities of defection within the equilibria and whether the game could be degenerate. The input variables are: 

103 

104 'payoff_matrix', a numpy array containing the payoffs obtained for each action of the game; and 

105 

106 'nash_equilibrium_algorithm', a string containing either "Support 

107 Enumeration", "Vertex Enumeration" or "Lemke Howson". This indicates which 

108 method will be used in calculating the Nash Equilibria. 

109 """ 

110 

111 game = nash.Game(payoff_matrix, payoff_matrix.transpose()) 

112 

113 with warnings.catch_warnings(record=True) as w: 

114 warnings.simplefilter("always") 

115 

116 if nash_equilibrium_algorithm == "Support Enumeration": 

117 nash_equilibria = list(game.support_enumeration()) 

118 # print(nash_equilibria) 

119 

120 elif nash_equilibrium_algorithm == "Vertex Enumeration": 

121 nash_equilibria = list(game.vertex_enumeration()) 

122 # print(nash_equilibria) 

123 

124 elif nash_equilibrium_algorithm == "Lemke Howson": 

125 nash_equilibria = list(game.lemke_howson_enumeration()) 

126 # print(nash_equilibria) 

127 

128 else: 

129 raise Exception( 

130 "nash_equilibrium_algorithm should be one of ['Support Enumeration', 'Vertex Enumeration', 'Lemke Howson']" 

131 ) 

132 

133 if len(w) == 0: 

134 perhaps_degenerate = False 

135 

136 else: 

137 perhaps_degenerate = "degenerate" in str(w[0].message) 

138 

139 prob_of_defection_in_equilibria = [sigma_1[-1] for sigma_1, _ in nash_equilibria] 

140 

141 least_prob_of_defection_in_equilibria = min(prob_of_defection_in_equilibria) 

142 greatest_prob_of_defection_in_equilibria = max(prob_of_defection_in_equilibria) 

143 

144 get_prob_of_defect_output_dict = { 

145 "nash equilibria": np.array(nash_equilibria), 

146 "least prob of defect": least_prob_of_defection_in_equilibria, 

147 "greatest prob of defect": greatest_prob_of_defection_in_equilibria, 

148 "could be degenerate": perhaps_degenerate, 

149 } 

150 

151 return get_prob_of_defect_output_dict
